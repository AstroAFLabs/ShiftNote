  <script>
    document.getElementById('jsonFileInput').addEventListener('change', function(event) {
        const file = event.target.files[0];
        const reader = new FileReader();
        reader.onload = function(e) {
            const jsonData = JSON.parse(e.target.result);
            sendJsonToServer(jsonData);
        };
        reader.readAsText(file);
    });
    
    function sendJsonToServer(jsonData) {
        const xhr = new XMLHttpRequest();
        xhr.open('POST', 'process_json_file.php', true);
        xhr.setRequestHeader('Content-Type', 'application/json;charset=UTF-8');
        xhr.onreadystatechange = function() {
            if (xhr.readyState === 4 && xhr.status === 200) {
                const responseData = JSON.parse(xhr.responseText);
                populateForm(responseData);
            }
        };
        xhr.send(JSON.stringify(jsonData));
    }
    
    function populateForm(data) {
        const form = document.getElementById('shiftForm');
        for (const key in data) {
            if (data.hasOwnProperty(key)) {
                const element = form.elements[key];
                if (element) {
                    if (element.type === 'checkbox' || element.type === 'radio') {
                        element.checked = data[key];
                    } else if (element.type === 'select-one' || element.type === 'select-multiple') {
                        for (let i = 0; i < element.options.length; i++) {
                            if (element.options[i].value == data[key]) {
                                element.selectedIndex = i;
                                element.options[i].style.backgroundColor = 'yellow'; // Highlight option
                                break;
                            }
                        }
                    } else {
                        element.value = data[key];
                    }
                }
            }
        }
    }
